CREATE TABLE USUARIO(
ID_USUARIO INT(10) NOT NULL,
PRIMARY KEY(ID_USUARIO),
NOMBRE VARCHAR(45),
APELLIDO VARCHAR(45),
DIRECCION VARCHAR(45),
TELEFONO INT(10),
EMAIL VARCHAR(45),
CIUDAD VARCHAR(45),
FECHA_DE_NACIMIENTO VARCHAR(45)
);

CREATE TABLE ROL(
ID_ROL INT(10) NOT NULL,
PRIMARY KEY(ID_ROL),
TIPO_ROL VARCHAR(45)
    
);


CREATE TABLE ADMINISTRADOR(
ID_ADMINISTRADOR INT(10) NOT NULL,
ID_USUARIO INT(10),
PRIMARY KEY(ID_ADMINISTRADOR),
NOMBRE VARCHAR(45),
APELLIDO VARCHAR(45),
DIRECCION VARCHAR(45),
TELEFONO INT(10),
EMAIL VARCHAR(45),
CIUDAD VARCHAR(45),
FOREIGN KEY(ID_USUARIO) REFERENCES USUARIO(ID_USUARIO)
);



CREATE TABLE SERVICIOS(
ID_SERVICIOS VARCHAR(10) NOT NULL,
PRIMARY KEY(ID_SERVICIOS),
NOMBRE_SERVICIOS VARCHAR(100),
PRECIO VARCHAR(100),
ID_CATEGORIA_SERVICIOS INT(100)
);





CREATE TABLE VENTAS(
ID_VENTAS INT(10) NOT NULL,
PRIMARY KEY(ID_VENTAS),
GANANCIAS VARCHAR(45),
PRODUCTOS_VENDIDOS VARCHAR(45),
SERVICIOS_REALIZADOS VARCHAR(45),
FECHA VARCHAR(45),
VALOR_TOTAL VARCHAR(45),
VALOR_SERVICIO VARCHAR(45),
VALOR_PRODUCTO VARCHAR(45),
ID_CLIENTE INT (10),
ID_VEHICULO INT(10),
ID_SERVICIOS INT(10),
FOREIGN KEY(ID_SERVICIOS)REFERENCES SERVICIOS (ID_SERVICIOS)
);



CREATE TABLE CATEGORIA_SERVICIOS(
ID_CATEGORIA_SERVICIOS VARCHAR(10) NOT NULL,
PRIMARY KEY(ID_CATEGORIA_SERVICIOS),
NOMBRE_CATEGORIA_SERVICIOS VARCHAR(45)
);


CREATE TABLE CATEGORIA_DE_PRODUCTOS(
    ID_CATEGORIA_DE_PRODUCTO INT(10) NOT NULL,
    PRIMARY KEY(ID_CATEGORIA_DE_PRODUCTO),
    NOMBRE_CATEGORIA_PRODUCTO VARCHAR(45)
);

CREATE TABLE PRODUCTOS(
    ID_PRODUCTOS INT(10) NOT NULL,
    PRIMARY KEY(ID_PRODUCTOS),
    NOMBRE_PRODUCTO VARCHAR(45),
    CANTIDAD_PRODUCTO VARCHAR(45),
    DESCRIPCION VARCHAR(45),
    ID_CATEGORIA_DE_PRODUCTOS INT(10),
    ID_VENTAS INT (10),
    FOREIGN KEY(ID_CATEGORIA_DE_PRODUCTOS) REFERENCES CATEGORIA_DE_PRODUCTOS(ID_CATEGORIA_DE_PRODUCTO),
    FOREIGN KEY(ID_VENTAS) REFERENCES VENTAS(ID_VENTAS)
);

CREATE TABLE ORDEN_TRABAJO(
ID_ORDEN_TRABAJO INT(10) NOT NULL,
PRIMARY KEY(ID_ORDEN_TRABAJO),
AÑADIR_ORDEN VARCHAR(45),
ASIGNAR_MECANICO VARCHAR(45),
EDITAR_ORDEN VARCHAR(45),
CONSULTAR_MATRICULO_VEHICULO VARCHAR(45),
ASIGNAR_TARIFA VARCHAR(45),
ID_MECANICOS INT(10),
ID_USUARIO INT(10)
);


CREATE TABLE ESTADO_ORDEN_TRABAJO(
ID_ESTADO_ORDEN_TRABAJO INT(10) NOT NULL,
PRIMARY KEY(ID_ESTADO_ORDEN_TRABAJO),
TERMINADO VARCHAR(45),
ATRASADO VARCHAR(45),
EN_ESPERA VARCHAR(45),
SIN_INICIAR VARCHAR(45),
ID_MECANICOS INT(10),
ID_USUARIO INT(10),
ID_ORDEN_TRABAJO INT(10)
);




CREATE TABLE MECANICOS(
ID_MECANICOS INT(10) NOT NULL,
PRIMARY KEY(ID_MECANICOS),
NOMBRE VARCHAR(45),
APELLIDO VARCHAR(45),
DIRECCION VARCHAR(45),
TELEFONO INT(10),
EMAIL VARCHAR(45),
CIUDAD VARCHAR(45),
ESPECIALIDAD VARCHAR(45),
ID_USUARIO INT(10)
);


CREATE TABLE CRONOGRAMA(
ID_CRONOGRAMA INT(10) NOT NULL,
PRIMARY KEY(ID_CRONOGRAMA),
FECHA_ENTREGA DATE,
ID_USUARIO INT(10),
ID_MECANICOS INT(10),
ID_ORDEN_TRABAJO INT(10),
);


CREATE TABLE GESTION_CLIENTES(
ID_GESTION_CLIENTES INT(10) NOT NULL,
PRIMARY KEY(ID_GESTION_CLIENTES),
AÑADIR_CLIENTE VARCHAR(45),
EDITAR_CLIENTE VARCHAR(45),
ELIMINAR_CLIENTE VARCHAR(45),
CONSULTAR_CLIENTE VARCHAR(45),
ID_CLIENTES INT(10), 
ID_USUARIO INT(10),
FOREIGN KEY(ID_USUARIO) REFERENCES USUARIO(ID_USUARIO),
FOREIGN KEY(ID_CLIENTES) REFERENCES CLIENTES(ID_CLIENTES)
);



CREATE TABLE VEHICULO(
ID_VEHICULO INT(10) NOT NULL,
PRIMARY KEY(ID_VEHICULO),
VIN VARCHAR(45),
MODELO VARCHAR(45),
COLOR VARCHAR(45)
);


CREATE TABLE PROVEEDOR(
ID_PROVEEDOR INT(10) NOT NULL,
PRIMARY KEY(ID_PROVEEDOR),
CONTACTO_PROVEEDOR VARCHAR(45),
EMPRESA_PROVEEDOR VARCHAR(45),
DIRECCION_PROVEEDOR VARCHAR(45),
EMAIL_PROVEEDOR VARCHAR(45),
CIUDAD_PROVEEDOR VARCHAR(45),
TELEFONO_PROVEEDOR VARCHAR(45)
);
